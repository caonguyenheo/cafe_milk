131 int16_t CdnClient_ConnectFileServer(CdnClient_t *pCdnClient, uint8_t *pFileUrl, int32_t SecuredConnection)
132 {
133  uint8_t ServerNameBuf[MAX_SERVER_NAME];
134  uint8_t *pReqUriBuf;
135  int16_t Status=0;
136 
137  /* extract CDN server name and file name */
138  pReqUriBuf = HttpClient_ParseUrl((uint8_t *)pFileUrl, ServerNameBuf, &Status);
139  if (pReqUriBuf == 0)
140  {
141  _SlOtaLibTrace(("CdnClient_ConnectFileServer: ERROR, HttpClient_ParseUrl, Status=%ld\r\n", Status));
142  return Status;
143  }
144 
145  /* connect to the CDN server */
146  pCdnClient->PortNum = SOCKET_PORT_DEFAULT;
147  pCdnClient->FileSockId = HttpClient_Connect(ServerNameBuf, 0, pCdnClient->PortNum, SecuredConnection, SOCKET_BLOCKING);
148  if (pCdnClient->FileSockId < 0)
149  {
150  _SlOtaLibTrace(("CdnClient_ConnectFileServer: ERROR on HttpClient_Connect, Status=%d\r\n", pCdnClient->FileSockId));
151  CdnClient_CloseServer(pCdnClient);
152  return pCdnClient->FileSockId;
153  }
154 
155  return CDN_STATUS_OK;
156 }
